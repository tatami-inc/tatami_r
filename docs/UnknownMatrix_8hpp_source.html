121 if (spacings.size() != 2) {
122 auto ctype = get_class_name(seed);
123 throw std::runtime_error("'chunkGrid(<" + ctype + ">)@spacings' should be an integer vector of length 2 with non-negative values");
124 }
125
126 auto populate = [](Index_ extent, Index_ spacing, std::vector<Index_>& map, std::vector<Index_>& ticks) {
127 if (spacing == 0) {
128 ticks.push_back(0);
129 } else {
130 ticks.reserve((extent / spacing) + (extent % spacing > 0) + 1);
131 Index_ start = 0;
132 ticks.push_back(start);
133 while (start != extent) {
134 auto to_fill = std::min(spacing, extent - start);
135 std::fill_n(map.begin() + start, to_fill, ticks.size() - 1);
136 start += to_fill;
137 ticks.push_back(start);
138 }
139 }
140 };
141
142 my_row_max_chunk_size = spacings[0];
143 populate(my_nrow, my_row_max_chunk_size, my_row_chunk_map, my_row_chunk_ticks);
144 my_col_max_chunk_size = spacings[1];
145 populate(my_ncol, my_col_max_chunk_size, my_col_chunk_map, my_col_chunk_ticks);
146
147 } else if (grid_cls == "ArbitraryArrayGrid") {
148 Rcpp::List ticks(Rcpp::RObject(grid.slot("tickmarks")));
149 if (ticks.size() != 2) {
150 auto ctype = get_class_name(seed);
151 throw std::runtime_error("'chunkGrid(<" + ctype + ">)@tickmarks' should return a list of length 2");
152 }
153
154 auto populate = [](Index_ extent, const Rcpp::IntegerVector& ticks, std::vector<Index_>& map, std::vector<Index_>& new_ticks, Index_& max_chunk_size) {
155 if (ticks.size() == 0 || ticks[ticks.size() - 1] != static_cast<int>(extent)) {
156 throw std::runtime_error("invalid ticks returned by 'chunkGrid'");
157 }
158 new_ticks.resize(ticks.size() + 1);
159 std::copy(ticks.begin(), ticks.end(), new_ticks.begin() + 1);
160
161 max_chunk_size = 0;
162 int start = 0;
163 map.resize(extent);
164 Index_ counter = 0;
165
166 for (auto t : ticks) {
167 if (t < start) {
168 throw std::runtime_error("invalid ticks returned by 'chunkGrid'");
169 }
170 Index_ to_fill = t - start;
171 if (to_fill > max_chunk_size) {
172 max_chunk_size = to_fill;
173 }
174 std::fill_n(map.begin() + start, to_fill, counter);
175 ++counter;
176 start = t;
177 }
178 };
179
180 Rcpp::IntegerVector first(ticks[0]);
181 populate(my_nrow, first, my_row_chunk_map, my_row_chunk_ticks, my_row_max_chunk_size);
182 Rcpp::IntegerVector second(ticks[1]);
183 populate(my_ncol, second, my_col_chunk_map, my_col_chunk_ticks, my_col_max_chunk_size);
184
185 } else {
186 auto ctype = get_class_name(seed);
187 throw std::runtime_error("instance of unknown class '" + grid_cls + "' returned by 'chunkGrid(<" + ctype + ">)");
188 }
189
190 // Choose the dimension that requires pulling out fewer chunks.
191 auto chunks_per_row = my_col_chunk_ticks.size() - 1;
192 auto chunks_per_col = my_row_chunk_ticks.size() - 1;
193 my_prefer_rows = chunks_per_row <= chunks_per_col;
194 }
195 }
196
197 my_cache_size_in_bytes = opt.maximum_cache_size;
198 my_require_minimum_cache = opt.require_minimum_cache;
199 if (my_cache_size_in_bytes == static_cast<size_t>(-1)) {
200 Rcpp::Function fun = my_delayed_env["getAutoBlockSize"];
201 Rcpp::NumericVector bsize = fun();
202 if (bsize.size() != 1 || bsize[0] < 0) {
203 throw std::runtime_error("'getAutoBlockSize()' should return a non-negative number of bytes");
204 }
205 my_cache_size_in_bytes = bsize[0];
206 }
207 }

224 // To decide how many chunks to store in the cache, we pretend the largest
225 // chunk is a good representative. This is a bit suboptimal for irregular
226 // chunks but the LruSlabCache class doesn't have a good way of dealing
227 // with this right now. The fundamental problem is that variable slabs will
228 // either (i) all reach the maximum allocation eventually, if slabs are
229 // reused, or (ii) require lots of allocations, if slabs are not reused, or
